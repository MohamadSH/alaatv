Resolution switcher

Code Setup

<!--Head tag CSS-->
<link href="../skins/nuevo/videojs.min.css" rel="stylesheet" type="text/css" />

<!--Multiple reslolution video on page setup-->
<video id="player_one" class="video-js vjs-fluid vjs-default-skin" controls preload="auto" poster="images/poster.jpg">
	<source src="//cdn.nuevolab.com/media/bmv/bmv_240.mp4" res="240" label="240p" type="video/mp4">
	<source src="//cdn.nuevolab.com/media/bmv/bmv_360.mp4" res="360" label="360p" type="video/mp4">
	<source src="//cdn.nuevolab.com/media/bmv/bmv_480.mp4" res="480" label="480p" default type="video/mp4">
	<source src="//cdn.nuevolab.com/media/bmv/bmv_720.mp4" res="720" label="720p" type="video/mp4">
</video>

<!--Load player and Nuevo plugin-->
<script type="text/javascript" src="../video.min.js"></script>
<script type="text/javascript" src="../nuevo.min.js"></script>

<!--Initialize player and plugin-->
<script>
	var player = videojs('player_one');
	player.nuevo({ 
		logotitle:'Nuevo plugin for Videojs',
		logo: 'images/logo.png',
		logoposition:'RT',
		logourl:'https://www.nuevolab.com'
	});
</script>
